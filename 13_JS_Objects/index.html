Object in JavaScript

        syntax:
            let object_name = {}

        let student = {
            name: "Ram",
            age: 20,
            course: "Btech"
        }

        console.log(student);
        console.log(student.name); // Ram - accessing using dot notation
        console.log(student.age); // 20 - accessing using array notation

For in loop through objects

        syntax: 
            for(let var in object_name);

        let bank = {
            "index": 0,
            "isActive": false,
            "balance": "$3,513.89",
            "picture": "http://placehold.it/32x32",
            "age": 22,
            "eyeColor": "green",
            "name": "Mcintosh Parrish",
            "gender": "male",
            "company": "SUREMAX",
            "email": "mcintoshparrish@suremax.com",
            "phone": "+1 (900) 560-3053",
            "city": 'Bangalore',
            "BankName": "SBI",
        }

        for (let key in bank){
            console.log(key);
        }

        for (let key in bank) {
            console.log(`${key} : ${bank[key]}`)
        }
